export function randomQuote(gender?: string): Quote {
  let pool: readonly Quote[];
  if (gender === 'f') {
    pool = [...QUOTES_FEMALE, ...QUOTES_FEMALE, ...QUOTES];
  } else if (gender === 'm') {
    pool = [...QUOTES_MALE, ...QUOTES_MALE, ...QUOTES];
  } else {
    pool = QUOTES;
  }
  return pool[Math.floor(Math.random() * pool.length)];
}
